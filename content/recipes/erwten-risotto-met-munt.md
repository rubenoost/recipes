ingredients:
  - item: "arborio rijst"
    amount: "125 gr"
  - item: "ui, gesnipperd"
    amount: "1"
  - item: "tuinerwten, gepureerd"
    amount: "50 gr"
  - item: "groente bouillon"
    amount: "500 ml"
  - item: "tuinerwten"
    amount: "150-200 gr"
  - item: "takje munt"
    amount: "1"
  - item: "boter"
    amount: ""
  - item: "olijfolie"
    amount: ""
  - item: "Parmigiano Reggiano"
    amount: "40 gr"
instructions:
  - step: "Smelt boter met olijfolie in de pan en bak de uien glazig (niet kleuren)."
  - step: "Voeg de rijst toe en goed roeren."
  - step: "Voeg beetje voor beetje de bouillon toe tot de rijst gaar is."
  - step: "Voeg het takje munt toe, roer vervolgens voorzichtig de erwten-puree en kaas erdoor tot helemaal gesmolten."
  - step: "Roer nu de erwten erdoor en verwijder de munt."
